CREATE VIEW `applicants_databases` AS
SELECT
    `source_setting`.`id` AS `id`,
    `source_setting`.`name` AS `name`,
    COUNT(`applicants`.`source_id`) AS `count`
FROM
    `source_setting`
LEFT JOIN
    `applicants`
ON
    `source_setting`.`id` = `applicants`.`source_id`
GROUP BY
    `source_setting`.`id`,
    `source_setting`.`name`;
